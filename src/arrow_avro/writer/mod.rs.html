18//! Avro writer implementation for the `arrow-avro` crate.
19//!
20//! # Overview
21//!
22//! *   Use **`AvroWriter`** (Object Container File) when you want a
23//!     self‑contained Avro file with header, schema JSON, optional compression,
24//!     blocks, and sync markers.
25//! *   Use **`AvroStreamWriter`** (raw binary stream) when you already know the
26//!     schema out‑of‑band (i.e., via a schema registry) and need a stream
27//!     of Avro‑encoded records with minimal framing.
28//!
29
30/// Encodes `RecordBatch` into the Avro binary format.
31pub mod encoder;
32/// Logic for different Avro container file formats.
33pub mod format;

44/// Builder to configure and create a `Writer`.
45#[derive(Debug, Clone)]
46pub struct WriterBuilder {
47    schema: Schema,
48    codec: Option<CompressionCodec>,
49}
50
51impl WriterBuilder {
52    /// Create a new builder with default settings.
53    pub fn new(schema: Schema) -> Self {
54        Self {
55            schema,
56            codec: None,
57        }
58    }
59
60    /// Change the compression codec.
61    pub fn with_compression(mut self, codec: Option<CompressionCodec>) -> Self {
62        self.codec = codec;
63        self
64    }
65
66    /// Create a new `Writer` with specified `AvroFormat` and builder options.
67    pub fn build<W, F>(self, writer: W) -> Writer<W, F>
68    where
69        W: Write,
70        F: AvroFormat,
71    {
72        Writer {
73            writer,
74            schema: Arc::from(self.schema),
75            format: F::default(),
76            compression: self.codec,
77            started: false,
78        }
79    }
80}

82/// Generic Avro writer.
83#[derive(Debug)]
84pub struct Writer<W: Write, F: AvroFormat> {
85    writer: W,
86    schema: Arc<Schema>,
87    format: F,
88    compression: Option<CompressionCodec>,
89    started: bool,
90}
91
92/// Alias for an Avro **Object Container File** writer.
93pub type AvroWriter<W> = Writer<W, AvroOcfFormat>;
94/// Alias for a raw Avro **binary stream** writer.
95pub type AvroStreamWriter<W> = Writer<W, AvroBinaryFormat>;
96
97impl<W: Write> Writer<W, AvroOcfFormat> {
98    /// Convenience constructor – same as
99    pub fn new(writer: W, schema: Schema) -> Result<Self, ArrowError> {
100        Ok(WriterBuilder::new(schema).build::<W, AvroOcfFormat>(writer))
101    }
102
103    /// Change the compression codec after construction.
104    pub fn with_compression(mut self, codec: Option<CompressionCodec>) -> Self {
105        self.compression = codec;
106        self
107    }
108
109    /// Return a reference to the 16‑byte sync marker generated for this file.
110    pub fn sync_marker(&self) -> Option<&[u8; 16]> {
111        self.format.sync_marker()
112    }
113}
114
115impl<W: Write> Writer<W, AvroBinaryFormat> {
116    /// Convenience constructor to create a new [`AvroStreamWriter`].
117    pub fn new(writer: W, schema: Schema) -> Result<Self, ArrowError> {
118        Ok(WriterBuilder::new(schema).build::<W, AvroBinaryFormat>(writer))
119    }
120}
121
122impl<W: Write, F: AvroFormat> Writer<W, F> {
123    /// Serialize one [`RecordBatch`] to the output.
124    pub fn write(&mut self, batch: &RecordBatch) -> Result<(), ArrowError> {
125        if !self.started {
126            self.format
127                .start_stream(&mut self.writer, &self.schema, self.compression)?;
128            self.started = true;
129        }
130        if batch.schema() != self.schema {
131            return Err(ArrowError::SchemaError(
132                "Schema of RecordBatch differs from Writer schema".to_string(),
133            ));
134        }
135        match self.format.sync_marker() {
136            Some(&sync) => self.write_ocf_block(batch, &sync),
137            None => self.write_stream(batch),
138        }
139    }
140
141    /// A convenience method to write a slice of [`RecordBatch`].
142    ///
143    /// This is equivalent to calling `write` for each batch in the slice.
144    pub fn write_batches(&mut self, batches: &[&RecordBatch]) -> Result<(), ArrowError> {
145        for b in batches {
146            self.write(b)?;
147        }
148        Ok(())
149    }
150
151    /// Flush remaining buffered data and (for OCF) ensure the header is present.
152    pub fn finish(&mut self) -> Result<(), ArrowError> {
153        if !self.started {
154            self.format
155                .start_stream(&mut self.writer, &self.schema, self.compression)?;
156            self.started = true;
157        }
158        self.writer
159            .flush()
160            .map_err(|e| ArrowError::IoError(format!("Error flushing writer: {e}"), e))
161    }
162
163    /// Consume the writer, returning the underlying output object.
164    pub fn into_inner(self) -> W {
165        self.writer
166    }
167
168    fn write_ocf_block(&mut self, batch: &RecordBatch, sync: &[u8; 16]) -> Result<(), ArrowError> {
169        let mut buf = Vec::<u8>::with_capacity(1024);
170        encode_record_batch(batch, &mut buf)?;
171        let encoded = match self.compression {
172            Some(codec) => codec.compress(&buf)?,
173            None => buf,
174        };
175        write_long(&mut self.writer, batch.num_rows() as i64)?;
176        write_long(&mut self.writer, encoded.len() as i64)?;
177        self.writer
178            .write_all(&encoded)
179            .map_err(|e| ArrowError::IoError(format!("Error writing Avro block: {e}"), e))?;
180        self.writer
181            .write_all(sync)
182            .map_err(|e| ArrowError::IoError(format!("Error writing Avro sync: {e}"), e))?;
183        Ok(())
184    }
185
186    fn write_stream(&mut self, batch: &RecordBatch) -> Result<(), ArrowError> {
187        encode_record_batch(batch, &mut self.writer)
188    }
189}

210    fn make_batch() -> RecordBatch {
211        let ids = Int32Array::from(vec![1, 2, 3]);
212        let names = BinaryArray::from_vec(vec![b"a".as_ref(), b"b".as_ref(), b"c".as_ref()]);
213        RecordBatch::try_new(
214            Arc::new(make_schema()),
215            vec![Arc::new(ids) as ArrayRef, Arc::new(names) as ArrayRef],

217        .expect("failed to build test RecordBatch")
218    }

220    fn contains_ascii(haystack: &[u8], needle: &[u8]) -> bool {
221        haystack.windows(needle.len()).any(|w| w == needle)
222    }

224    #[test]
225    fn test_ocf_writer_generates_header_and_sync() -> Result<(), ArrowError> {
226        let batch = make_batch();
227        let buffer: Vec<u8> = Vec::new();
228        let mut writer = AvroWriter::new(buffer, make_schema())?;
229        writer.write(&batch)?;
230        writer.finish()?;
231        let out = writer.into_inner();
232        assert_eq!(&out[..4], b"Obj\x01", "OCF magic bytes missing/incorrect");
233        let sync = AvroWriter::new(Vec::new(), make_schema())?
234            .sync_marker()
235            .cloned();
236        let trailer = &out[out.len() - 16..];
237        assert_eq!(trailer.len(), 16, "expected 16‑byte sync marker");
238        let _ = sync;
239        Ok(())
240    }

290    #[test]
291    fn test_roundtrip_alltypes_roundtrip_writer() -> Result<(), ArrowError> {
292        let files = [
293            "avro/alltypes_plain.avro",
294            "avro/alltypes_plain.snappy.avro",
295            "avro/alltypes_plain.zstandard.avro",
296            "avro/alltypes_plain.bzip2.avro",
297            "avro/alltypes_plain.xz.avro",
298        ];
299        for rel in files {
300            let path = arrow_test_data(rel);
301            let rdr_file = File::open(&path).expect("open input avro");
302            let mut reader = ReaderBuilder::new()
303                .build(BufReader::new(rdr_file))
304                .expect("build reader");
305            let schema = reader.schema();
306            let input_batches = reader.collect::<Result<Vec<_>, _>>()?;
307            let original =
308                arrow::compute::concat_batches(&schema, &input_batches).expect("concat input");
309            let tmp = NamedTempFile::new().expect("create temp file");
310            let out_path = tmp.into_temp_path();
311            let out_file = File::create(&out_path).expect("create temp avro");
312            let mut writer = AvroWriter::new(out_file, original.schema().as_ref().clone())?;
313            if rel.contains(".snappy.") {
314                writer = writer.with_compression(Some(CompressionCodec::Snappy));
315            } else if rel.contains(".zstandard.") {
316                writer = writer.with_compression(Some(CompressionCodec::ZStandard));
317            } else if rel.contains(".bzip2.") {
318                writer = writer.with_compression(Some(CompressionCodec::Bzip2));
319            } else if rel.contains(".xz.") {
320                writer = writer.with_compression(Some(CompressionCodec::Xz));
321            }
322            writer.write(&original)?;
323            writer.finish()?;
324            drop(writer);
325            let rt_file = File::open(&out_path).expect("open roundtrip avro");
326            let mut rt_reader = ReaderBuilder::new()
327                .build(BufReader::new(rt_file))
328                .expect("build roundtrip reader");
329            let rt_schema = rt_reader.schema();
330            let rt_batches = rt_reader.collect::<Result<Vec<_>, _>>()?;
331            let roundtrip =
332                arrow::compute::concat_batches(&rt_schema, &rt_batches).expect("concat roundtrip");
333            assert_eq!(
334                roundtrip, original,
335                "Round-trip batch mismatch for file: {}",
336                rel
337            );
338        }
339        Ok(())
340    }
